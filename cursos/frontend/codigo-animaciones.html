Logo animado

Css: 

              .animation-logo-img .logo-c {
                animation-name: logo-c-in;
                animation-duration : 2s;
                animation-iteration-count : 1;
                animation-direction: alternate;
              }

              .animation-logo-img .logo-z {
                opacity : 0;
              }
              
              .animation-logo-img .logo-z.active-animation {
                animation-name: logo-z;
                animation-duration: 1s;
                animation-iteration-count: 1;
                animation-fill-mode: forwards;
              }

              /*Configurando animación*/

              @keyframes logo-c-in{
                0% {
                  opacity: 0;
                  transform: translateX(150px) scale(.4) rotate(360deg);
                }
                50% {
                  transform: translateX(150px) scale(1) rotate(0);
                }
                75% {
                  opacity: 1;
                  transform: translateX(0px) scale(1) rotate(0);
                }
                100% {
                  opacity: 1;
                  transform: translateX(0) scale(1) rotate(0);
                }
              }

              @keyframes logo-z {
                0%
                  opacity: 0;
                  transform: translateX(-150px);
                }
                80% {
                  opacity: .8;
                  transform: translateX(50px) rotate(8deg);
                }
                100% {
                  opacity: 1;
                  transform: translateX(0) rotate(0);
                }
              }

Javascript: 

              const $logoC = document.getElementById('logo-c');
              const $logoZ = document.getElementById('logo-z');

              $logoC.addEventListener('animationend', () => {
                $logoZ.classList.add('active-animation');
              });

Javascript: 

              const $btnModal = document.getElementById('btn-modal');
              const $modal = document.getElementById('modal');
              const $closeModal = document.getElementById('close-modal');

              $btnModal.addEventListener('click', () => {
                $modal.classList.remove('out');
                $modal.classList.add('active');
              })
            
              $closeModal.addEventListener('click', () => {
                $modal.classList.add('out');
                $modal.classList.remove('active');
              });
